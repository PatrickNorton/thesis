\newcommand\Part[1]{
	\setcounter{r}{1}
	\foreach \x in {#1}{
			{\ifnum\value{r}=1
						\draw (0,\value{r}-1)--(\x,\value{r}-1);
					\fi}
			\draw (0,\value{r}) to (\x,\value{r});
			\foreach \y in {0, ..., \x} {\draw (\y,\value{r})--(\y,\value{r}-1);}
			\addtocounter{r}{1}
		}}
